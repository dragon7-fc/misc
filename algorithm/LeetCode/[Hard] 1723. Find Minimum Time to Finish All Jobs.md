1723. Find Minimum Time to Finish All Jobs

You are given an integer array `jobs`, where `jobs[i]` is the amount of time it takes to complete the `i`th job.

There are `k` workers that you can assign jobs to. Each job should be assigned to **exactly** one worker. The **working time** of a worker is the sum of the time it takes to complete all jobs assigned to them. Your goal is to devise an optimal assignment such that the **maximum working time** of any worker is **minimized**.

Return the **minimum** possible **maximum working time** of any assignment.

 

**Example 1:**
```
Input: jobs = [3,2,3], k = 3
Output: 3
Explanation: By assigning each person one job, the maximum time is 3.
```

**Example 2:**
```
Input: jobs = [1,2,4,7,8], k = 2
Output: 11
Explanation: Assign the jobs the following way:
Worker 1: 1, 2, 8 (working time = 1 + 2 + 8 = 11)
Worker 2: 4, 7 (working time = 4 + 7 = 11)
The maximum working time is 11.
```

**Constraints:**

* `1 <= k <= jobs.length <= 12`
* `1 <= jobs[i] <= 10^7`

**Solution 1: (Binary Search)**

This code basically reuses the exact code from #698 Partition to K Equal Sum Subsets.

The basic idea is you binary search for a target answer while using the bucket method to see if it's possible.
Since we want to find the lowest/leftmost possible answer, we use lo + hi >> 1, and lo = mid + 1

Optimizations:

1. Place the larger numbers first, as smaller numbers could be placed into a lot of buckets
1. If filling it in this bucket doesnt work, then nothing will work. This is bc we fill previous buckets before trying the next

```
Runtime: 60 ms
Memory Usage: 14.1 MB
```
```python
class Solution:
    def minimumTimeRequired(self, jobs: List[int], k: int) -> int:
        jobs.sort(reverse=True) # optimization (1)
        
        def possible(target, buckets, idx):
            if idx == len(jobs):
                return True
            for i in range(len(buckets)):
                buckets[i] += jobs[idx]
                if buckets[i] <= target and possible(target, buckets, idx+1):
                    return True
                buckets[i] -= jobs[idx]
                if buckets[i] == 0: # optimization (2)
                    break
            return False  
        
        lo = max(jobs)
        hi = sum(jobs)
        while lo < hi:
            mid = lo + hi >> 1
            b = [0] * k
            if not possible(mid, b, 0):
                lo = mid + 1
            else:
                hi = mid
        return lo
```

```c++
class Solution {
    // O(k * 3^n)
    bool check(int mid, vector<int> &presum, vector<int> &jobs, int k) {
        int n = jobs.size();
        const int FULL = (1 << n) - 1;
        vector<vector<bool>> dp(k + 1, vector<bool>(1 << n));
        dp[0][0] = true;
        for (int i = 1; i <= k; i ++) {
            for (int mask = 0; mask <= FULL; mask ++) {
                if (dp[i - 1][mask] == false) {
                    continue;
                }
                dp[i][mask] = true;
                int remain = FULL ^ mask;
                for (int sub = remain; sub; sub = (sub - 1) & remain) {
                    if (presum[sub] <= mid) {
                        dp[i][mask | sub] = true;
                    }
                }
            }
        }
        return dp[k][FULL];
    }
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        int left = *max_element(jobs.begin(), jobs.end());
        int right = accumulate(jobs.begin(), jobs.end(), 0);
        int n = jobs.size(), mid, ans;
        vector<int> presum(1 << n);
        for (int mask = 1; mask < (1 << n); mask ++) {
            int bit = mask & (-mask);
            int idx = __builtin_ctz(bit);
            presum[mask] = presum[mask ^ bit] + jobs[idx];
        }
        while (left <= right) {
            mid = left + (right - left) / 2;
            if (!check(mid, presum, jobs, k)) {
                left = mid + 1;
            } else {
                ans = mid;
                right = mid - 1;
            }
        }
        return ans;
    }
};
```

```c++
class Solution {
    // O(k * n)
    bool dfs(int idx, vector<int>& jobs, vector<int>& load, int limit) {
        if (idx == jobs.size())
            return true;
        int cur = jobs[idx];
        for (int w = 0; w < load.size(); w++) {
            if (load[w] + cur > limit) {
                continue;
            }
            load[w] += cur;
            if (dfs(idx + 1, jobs, load, limit)) {
                return true;
            }
            load[w] -= cur;
            if (load[w] == 0) {
                break;
            }
        }

        return false;
    }
public:
    int minimumTimeRequired(vector<int>& jobs, int k) {
        // Large jobs create conflicts first
        sort(jobs.rbegin(), jobs.rend());
        int left = *max_element(jobs.begin(), jobs.end());
        int right = accumulate(jobs.begin(), jobs.end(), 0);
        int ans = right;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            vector<int> load(k, 0);
            if (dfs(0, jobs, load, mid)) {
                ans = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        return ans;
    }
};
```
