1323. Maximum 69 Number

Given a positive integer `num` consisting only of digits 6 and 9.

Return the maximum number you can get by changing at most one digit (6 becomes 9, and 9 becomes 6).

 

**Example 1:**
```
Input: num = 9669
Output: 9969
Explanation: 
Changing the first digit results in 6669.
Changing the second digit results in 9969.
Changing the third digit results in 9699.
Changing the fourth digit results in 9666. 
The maximum number is 9969.
```

**Example 2:**
```
Input: num = 9996
Output: 9999
Explanation: Changing the last digit 6 to 9 results in the maximum number.
```

**Example 3:**
```
Input: num = 9999
Output: 9999
Explanation: It is better not to apply any change.
```

**Constraints:**

* `1 <= num <= 10^4`
* `num`'s digits are `6` or `9`.

```python
class Solution:
    def maximum69Number (self, num: int) -> int:
        # Since we start with the lowest digit, initialize curr_digit = 0.
        curr_digit = 0
        index_first_six = -1
        num_copy = num
        
        # Check every digit of 'num_copy' from low to high.
        while num_copy:
            # If the current digit is '6', record it as the highest digit of 6.
            if num_copy % 10 == 6:
                index_first_six = curr_digit
            
            # Move on to the next digit.
            num_copy //= 10
            curr_digit += 1
        
        # If we don't find any digit of '6', return the original number,
        # otherwise, increment 'num' by the difference made by the first '6'.
        return num if index_first_six == -1 else num + 3 * 10 ** index_first_six
```

```c++
class Solution {
public:
    int maximum69Number (int num) {
        string s = to_string(num);
        for (int i = 0; i < s.length(); i ++) {
            if (s[i] == '6') {
                s[i] = '9';
                break;
            }
        }
        return stoi(s);
    }
};
```

```c++
class Solution {

    int maximum69Number (int num) {
        int b = 1, r, ans = num;
        while (b*10 < num) {
            b *= 10;
        }
        r = num;
        while (r) {
            if ((r/b)%10 == 6) {
                ans = (num/b + 3) * b + r%b;
                break;
            }
            r %= b;
            b /= 10;
        }
        return ans;
    }
};
```
